task.spawn(function()
    while IntroActive do
        if not IntroActive then break end
        local Particle = Instance.new("Frame")
        Particle.Size = UDim2.new(0, math.random(4, 8), 0, math.random(4, 8))
        Particle.Position = UDim2.new(math.random(0, 100)/100, 0, -0.05, 0)
        Particle.BackgroundColor3 = Color3.fromRGB(255, math.random(50, 150), 0) -- Fire colors
        Particle.BorderSizePixel = 0
        Particle.ZIndex = 999999
        Particle.Parent = IntroBackground
        
        local UICorner = Instance.new("UICorner")
        UICorner.CornerRadius = UDim.new(1, 0)
        UICorner.Parent = Particle
        
        table.insert(FallingParticles, Particle)
        
        local FallSpeed = math.random(2, 4)
        local FallTween = TweenService:Create(Particle, TweenInfo.new(FallSpeed, Enum.EasingStyle.Linear), {
            Position = UDim2.new(Particle.Position.X.Scale + (math.random(-10, 10)/100), 0, 1.05, 0),
            BackgroundTransparency = 0.5
        })
        FallTween:Play()
        
        task.delay(FallSpeed, function()
            Particle:Destroy()
        end)
        
        task.wait(0.08)
    end
end)

local function CreateESP(Player)
    if ESPObjects[Player] then return end
    local Objects = {
        Box = Drawing.new("Square"),
        Line = Drawing.new("Line"),
        Name = Drawing.new("Text"),
        Skeleton = {}
    }
    Objects.Box.Thickness = 1.5
    Objects.Box.Filled = false
    Objects.Box.Visible = false
    Objects.Line.Thickness = 1.5
    Objects.Line.Visible = false
    Objects.Name.Size = 16
    Objects.Name.Center = true
    Objects.Name.Outline = true
    Objects.Name.Visible = false
    
    for i = 1, 6 do 
        local L = Drawing.new("Line")
        L.Thickness = 1.5
        L.Visible = false
        table.insert(Objects.Skeleton, L)
    end
    ESPObjects[Player] = Objects
end

local function RemoveESP(Player)
    if ESPObjects[Player] then
        ESPObjects[Player].Box:Remove()
        ESPObjects[Player].Line:Remove()
        ESPObjects[Player].Name:Remove()
        for _, Bone in pairs(ESPObjects[Player].Skeleton) do Bone:Remove() end
        ESPObjects[Player] = nil
    end
end

Players.PlayerAdded:Connect(CreateESP)
Players.PlayerRemoving:Connect(RemoveESP)
for _, p in pairs(Players:GetPlayers()) do if p ~= LocalPlayer then CreateESP(p) end end

local hue = 0
RunService.RenderStepped:Connect(function()
    hue = hue + 0.005 if hue > 1 then hue = 0 end
    
    local CurrentColor = Config.ESPRGB and Color3.fromHSV(hue, 1, 1) or Config.ESPColor
    local CurrentHitboxColor = Config.HitboxRGB and Color3.fromHSV(hue, 1, 1) or Config.HitboxColor
    
    for Player, ESP in pairs(ESPObjects) do
        if Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") and Player.Character:FindFirstChild("Humanoid") and Player.Character.Humanoid.Health > 0 then
            local Char = Player.Character
            local HRP = Char.HumanoidRootPart
            local Head = Char:FindFirstChild("Head")
            
            if Config.HitboxEnabled and Config.HitboxSize > 2 then
                HRP.Size = Vector3.new(Config.HitboxSize, Config.HitboxSize, Config.HitboxSize)
                HRP.Color = CurrentHitboxColor
                HRP.Transparency = 0.6
                HRP.Material = Enum.Material.Neon
                HRP.CanCollide = false
                if Head then Head.Transparency = Config.HideHead and 1 or 0 end
            else
                HRP.Size = Vector3.new(2, 2, 1)
                HRP.Transparency = 1
                if Head then Head.Transparency = 0 end
            end
            
            local HRP_Pos, OnScreen = Camera:WorldToViewportPoint(HRP.Position)
            
            if OnScreen then
                if Config.BoxESP then
                    local Head_Pos = Camera:WorldToViewportPoint(Head.Position + Vector3.new(0, 0.5, 0))
                    local Leg_Pos = Camera:WorldToViewportPoint(HRP.Position - Vector3.new(0, 3, 0))
                    local Height = math.abs(Head_Pos.Y - Leg_Pos.Y)
                    local Width = Height / 1.5
                    ESP.Box.Size = Vector2.new(Width, Height)
                    ESP.Box.Position = Vector2.new(HRP_Pos.X - Width / 2, Head_Pos.Y)
                    ESP.Box.Color = CurrentColor
                    ESP.Box.Visible = true
                else ESP.Box.Visible = false end
                
                if Config.LinesESP then
                    ESP.Line.From = Vector2.new(Camera.ViewportSize.X / 2, 0) 
                    ESP.Line.To = Vector2.new(HRP_Pos.X, HRP_Pos.Y)
                    ESP.Line.Color = CurrentColor
                    ESP.Line.Visible = true
                else ESP.Line.Visible = false end
                
                if Config.NameESP then
                    ESP.Name.Text = Player.Name
                    ESP.Name.Position = Vector2.new(HRP_Pos.X, Camera:WorldToViewportPoint(Head.Position + Vector3.new(0, 2, 0)).Y)
                    ESP.Name.Color = CurrentColor
                    ESP.Name.Visible = true
                else ESP.Name.Visible = false end
                
                if Config.SkeletonESP and Char:FindFirstChild("LeftUpperArm") and Char:FindFirstChild("RightUpperArm") and Char:FindFirstChild("LeftUpperLeg") and Char:FindFirstChild("RightUpperLeg") then
                    local Joints = {
                        {Head, Char.UpperTorso}, {Char.UpperTorso, Char.LowerTorso},
                        {Char.UpperTorso, Char.LeftUpperArm}, {Char.UpperTorso, Char.RightUpperArm},
                        {Char.LowerTorso, Char.LeftUpperLeg}, {Char.LowerTorso, Char.RightUpperLeg}
                    }
                    for i, Joint in pairs(Joints) do
                        local p1 = Camera:WorldToViewportPoint(Joint[1].Position)
                        local p2 = Camera:WorldToViewportPoint(Joint[2].Position)
                        ESP.Skeleton[i].From = Vector2.new(p1.X, p1.Y)
                        ESP.Skeleton[i].To = Vector2.new(p2.X, p2.Y)
                        ESP.Skeleton[i].Color = CurrentColor
                        ESP.Skeleton[i].Visible = true
                    end
                else
                    for _, Bone in pairs(ESP.Skeleton) do Bone.Visible = false end
                end
            else
                ESP.Box.Visible = false ESP.Line.Visible = false ESP.Name.Visible = false
                for _, Bone in pairs(ESP.Skeleton) do Bone.Visible = false end
            end
        else
            if ESPObjects[Player] then
                ESP.Box.Visible = false ESP.Line.Visible = false ESP.Name.Visible = false
                for _, Bone in pairs(ESP.Skeleton) do Bone.Visible = false end
            end
        end
    end
end)
